\begin{axis}[

    xlabel={Path Length $L$ [fm]},
    ylabel={Energy Loss $\langle\Delta E\rangle$ [GeV]},
    xmin=0, xmax=10,
    ymin=0, ymax=25,
    grid=major,
    grid style={gridcolor},
    legend pos=north west,
    legend style={fill=white, fill opacity=0.9, draw=none, font=\small},
    xlabel style={font=\large},
    ylabel style={font=\large},
    tick label style={font=\small},

]

% BDMPS-Z radiative loss (L^2)
\addplot[hotphase, ultra thick, solid, name path=rad]
    table [x index=0, y index=1] {data/physics_connections/energy_loss_vs_L.dat};
\addlegendentry{Radiative: $\langle\Delta E\rangle_{\mathrm{rad}} \propto \hat{q}L^2$}

% Collisional loss (L)
\addplot[coolphase, ultra thick, dashed, name path=coll]
    table [x index=0, y index=2] {data/physics_connections/energy_loss_vs_L.dat};
\addlegendentry{Collisional: $\langle\Delta E\rangle_{\mathrm{coll}} \propto L$}

% Total loss
\addplot[primaryblue, ultra thick, dotted, name path=total]
    table [x index=0, y index=3] {data/physics_connections/energy_loss_vs_L.dat};
\addlegendentry{Total: $\langle\Delta E\rangle_{\mathrm{tot}}$}

% Fill between radiative and total
\addplot[hotphase, opacity=0.15] fill between[of=rad and coll];

% System markers
\draw[gray, dashed, thick] (axis cs:2.5, 0) -- (axis cs:2.5, 25);
\node[font=\small, gray, rotate=90] at (axis cs:2.2, 15) {O--O};

\node[font=\small, gray, rotate=90] at (axis cs:5.7, 15) {Pb--Pb};

\node[font=\scriptsize, accentpurple, align=center] at (axis cs:4.5, 3.5) {Collisional\\dominates\\at small $L$};

\draw[accentpurple, thick, ->] (axis cs:7, 15) -- (axis cs:8, 18);
\node[font=\scriptsize, accentpurple, align=center] at (axis cs:6, 13) {Radiative\\dominates\\at large $L$};

% Physics box

    $\omega_c = \frac{1}{2}\hat{q}L^2$\\[3pt]

    (central Pb--Pb)
};

% Small system insight
\node[draw=secondaryorange, thick, fill=highlightcolor!30, rounded corners=2pt,
